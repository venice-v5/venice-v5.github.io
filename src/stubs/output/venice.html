venice

Venice is an open-source Micropython runtime for VEX V5 robots.

from venice import Motor, Gearset, Direction
import venice.vasyncio

async def main():
    my_motor = Motor(
        1,
        Direction.FORWARD,
        Gearset.GREEN
    )
    my_motor.set_voltage(10.0)
vasyncio.run(main())
